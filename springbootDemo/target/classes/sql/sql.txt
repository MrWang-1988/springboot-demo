create table EN_EMPLOYEE(
	EMPID VARCHAR(36) PRIMARY KEY COMMENT '员工ID',
	EMPNAME VARCHAR(30) NOT NULL COMMENT '员工姓名',
	GENDER VARCHAR(1) COMMENT '性别：F:女,M:男',
	PHONE VARCHAR(11) COMMENT '手机号',
	ENTRYDAY DATE COMMENT '入职时间',
	DEPT_ID VARCHAR(36) COMMENT '部门ID',
	CREATETIME TIMESTAMP COMMENT '创建时间'
);

create table EN_DEPT(
	DEPTID VARCHAR(36) PRIMARY KEY COMMENT '部门ID',
	DEPT_NAME VARCHAR(30) NOT NULL COMMENT '部门名称',
	CREATEDATE TIMESTAMP COMMENT '成立时间'
);
